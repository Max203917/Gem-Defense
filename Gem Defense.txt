<canvas id="game" width="800" height="600"></canvas>

<script>
const canvas = document.getElementById("game");
const ctx = canvas.getContext("2d");

const center = {x:400,y:300};

const scoreText = document.getElementById("score");
const mainMenu = document.getElementById("mainMenu");
const gameOverMenu = document.getElementById("gameOverMenu");

let score = 0;
let lives = 5;
let gameRunning = false;

const targets = {
    up:{x:400,y:180},
    down:{x:400,y:420},
    left:{x:280,y:300},
    right:{x:520,y:300}
};

let projectiles = [];

function resetGame(){
    score = 0;
    lives = 5;
    projectiles = [];
    scoreText.innerText = "Очки: 0";
}

function startGame(){
    resetGame();
    gameRunning = true;
    mainMenu.style.display = "none";
    gameOverMenu.style.display = "none";
    scoreText.style.display = "block";
}

function restartGame(){
    startGame();
}

function backToMenu(){
    gameRunning = false;
    mainMenu.style.display = "flex";
    gameOverMenu.style.display = "none";
    scoreText.style.display = "none";
    resetGame();
}

function endGame(){
    gameRunning = false;
    gameOverMenu.style.display = "flex";
}

function spawnProjectile(){
    if(!gameRunning) return;

    const dirs = ["up","down","left","right"];
    const dir = dirs[Math.floor(Math.random()*4)];

    let x,y,vx=0,vy=0;

    if(dir==="up"){ x=400; y=-20; vy=3; }
    if(dir==="down"){ x=400; y=620; vy=-3; }
    if(dir==="left"){ x=-20; y=300; vx=3; }
    if(dir==="right"){ x=820; y=300; vx=-3; }

    projectiles.push({x,y,vx,vy,dir});
}

setInterval(spawnProjectile,1000);

function drawDiamond(){
    ctx.fillStyle="purple";
    ctx.beginPath();
    ctx.moveTo(center.x,center.y-40);
    ctx.lineTo(center.x+40,center.y);
    ctx.lineTo(center.x,center.y+40);
    ctx.lineTo(center.x-40,center.y);
    ctx.closePath();
    ctx.fill();
}

function drawTargets(){
    ctx.strokeStyle="white";
    ctx.lineWidth=4;

    for(let key in targets){
        let t=targets[key];
        ctx.beginPath();
        ctx.arc(t.x,t.y,25,0,Math.PI*2);
        ctx.stroke();
    }
}

function drawLives(){
    for(let i=0;i<lives;i++){
        ctx.fillStyle="red";
        ctx.fillRect(700+i*18,20,15,15);
    }
}

function loseLife(){
    lives--;

    if(lives<=0){
        endGame();
    }
}

function updateProjectiles(){
    if(!gameRunning) return;

    ctx.fillStyle="blue";

    for(let i=projectiles.length-1;i>=0;i--){
        let p=projectiles[i];

        p.x += p.vx;
        p.y += p.vy;

        ctx.beginPath();
        ctx.arc(p.x,p.y,20,0,Math.PI*2);
        ctx.fill();

        let dx=p.x-center.x;
        let dy=p.y-center.y;
        let dist=Math.sqrt(dx*dx+dy*dy);

        if(dist<50){
            projectiles.splice(i,1);
            loseLife();
        }
    }
}

function hit(direction){
    if(!gameRunning) return;

    for(let i=0;i<projectiles.length;i++){
        let p=projectiles[i];
        let t=targets[direction];

        let dx=p.x-t.x;
        let dy=p.y-t.y;
        let dist=Math.sqrt(dx*dx+dy*dy);

        if(dist<25 && p.dir===direction){
            projectiles.splice(i,1);
            score++;
            scoreText.innerText="Очки: "+score;
            return;
        }
    }
}

document.addEventListener("keydown",(e)=>{
    if(e.key==="ArrowUp") hit("up");
    if(e.key==="ArrowDown") hit("down");
    if(e.key==="ArrowLeft") hit("left");
    if(e.key==="ArrowRight") hit("right");
});

function gameLoop(){
    ctx.clearRect(0,0,800,600);

    drawDiamond();
    drawTargets();
    drawLives();
    updateProjectiles();

    requestAnimationFrame(gameLoop);
}

gameLoop();
</script>
